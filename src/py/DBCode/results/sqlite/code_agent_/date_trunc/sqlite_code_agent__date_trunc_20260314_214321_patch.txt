+**
+** Truncate the timestamp to the specified precision.
+** UNIT must be one of: 'year', 'month', 'day', 'hour', 'minute', 'second'.
+** Returns the truncated timestamp as a Julian day number.
+*/
+static void dateTruncFunc(
+  sqlite3_context *context,
+  int argc,
+  sqlite3_value **argv
+){
+  const unsigned char *zUnit;
+  DateTime x;
+  int rc;
+
+  /* Check argument count */
+  if( argc!=2 ){
+    sqlite3_result_error(context, "date_trunc() requires exactly 2 arguments", -1);
+    return;
+  }
+
+  /* Get and validate the unit argument */
+  zUnit = sqlite3_value_text(argv[0]);
+  if( zUnit==0 ){
+    sqlite3_result_null(context);
+    return;
+  }
+
+  /* Parse the timestamp */
+  memset(&x, 0, sizeof(x));
+  rc = isDate(context, 1, &argv[1], &x);
+  if( rc ){
+    sqlite3_result_error(context, "date_trunc: isDate failed", -1);
+    return;
+  }
+
+  /* Convert to YMD and HMS */
+  computeYMD_HMS(&x);
+  if( x.isError ){
+    sqlite3_result_null(context);
+    return;
+  }
+
+  /* Truncate based on unit */
+  zUnit = sqlite3_value_text(argv[0]);
+  if( sqlite3_stricmp((const char*)zUnit, "year")==0 ){
+    x.M = 1;
+    x.D = 1;
+    x.h = 0;
+    x.m = 0;
+    x.s = 0.0;
+  }else if( sqlite3_stricmp((const char*)zUnit, "month")==0 ){
+    x.D = 1;
+    x.h = 0;
+    x.m = 0;
+    x.s = 0.0;
+  }else if( sqlite3_stricmp((const char*)zUnit, "day")==0 ){
+    x.h = 0;
+    x.m = 0;
+    x.s = 0.0;
+  }else if( sqlite3_stricmp((const char*)zUnit, "hour")==0 ){
+    x.m = 0;
+    x.s = 0.0;
+  }else if( sqlite3_stricmp((const char*)zUnit, "minute")==0 ){
+    x.s = 0.0;
+  }else if( sqlite3_stricmp((const char*)zUnit, "second")==0 ){
+    /* No truncation needed - already at second precision */
+  }else{
+    sqlite3_result_error(context, "date_trunc(): invalid unit", -1);
+    return;
+  }
+

+  x.validJD = 0;
+  computeJD(&x);
+  if( x.isError || !validJulianDay(x.iJD) ){
+    sqlite3_result_null(context);
+    return;
+  }
+
+  /* Return as Julian day number */
+  sqlite3_result_double(context, x.iJD/86400000.0);
