+/*
+**    date_trunc( UNIT, TIMESTRING, MOD, MOD, ...)
+**
+** Truncate a timestamp to the specified precision.
+** UNIT can be: 'year', 'month', 'day', 'hour', 'minute', 'second'
+** Returns the truncated timestamp as YYYY-MM-DD HH:MM:SS
+*/
+static void dateTruncFunc(
+  sqlite3_context *context,
+  int argc,
+  sqlite3_value **argv
+){
+  DateTime x;
+  const char *zUnit;
+  
+  if( argc < 2 ) return;
+  
+  zUnit = (const char*)sqlite3_value_text(argv[0]);
+  if( zUnit==0 ) return;
+  
+  if( isDate(context, argc-1, argv+1, &x) ) return;
+  
+  computeJD(&x);
+  computeYMD_HMS(&x);
+  
+  if( sqlite3_stricmp(zUnit, "year")==0 ){
+    x.M = 1;
+    x.D = 1;
+    x.h = 0;
+    x.m = 0;
+    x.s = 0.0;
+  }else if( sqlite3_stricmp(zUnit, "month")==0 ){
+    x.D = 1;
+    x.h = 0;
+    x.m = 0;
+    x.s = 0.0;
+  }else if( sqlite3_stricmp(zUnit, "day")==0 ){
+    x.h = 0;
+    x.m = 0;
+    x.s = 0.0;
+  }else if( sqlite3_stricmp(zUnit, "hour")==0 ){
+    x.m = 0;
+    x.s = 0.0;
+  }else if( sqlite3_stricmp(zUnit, "minute")==0 ){
+    x.s = 0.0;
+  }else if( sqlite3_stricmp(zUnit, "second")==0 ){
+    /* No truncation needed for seconds */
+  }else{
+    sqlite3_result_error(context, "date_trunc: unknown unit", -1);
+    return;
+  }
+  
+  x.validJD = 0;
+  computeJD(&x);
+  if( x.isError || !validJulianDay(x.iJD) ){
+    return;
+  }
+  computeYMD_HMS(&x);
+  
+  /* Output as YYYY-MM-DD HH:MM:SS */
+  {
+    int Y, s;
+    char zBuf[32];
+    Y = x.Y;
+    if( Y<0 ) Y = -Y;
+    zBuf[1] = '0' + (Y/1000)%10;
+    zBuf[2] = '0' + (Y/100)%10;
+    zBuf[3] = '0' + (Y/10)%10;
+    zBuf[4] = '0' + (Y)%10;
+    zBuf[5] = '-';
+    zBuf[6] = '0' + (x.M/10)%10;
+    zBuf[7] = '0' + (x.M)%10;
+    zBuf[8] = '-';
+    zBuf[9] = '0' + (x.D/10)%10;
+    zBuf[10] = '0' + (x.D)%10;
+    zBuf[11] = ' ';
+    zBuf[12] = '0' + (x.h/10)%10;
+    zBuf[13] = '0' + (x.h)%10;
+    zBuf[14] = ':';
+    zBuf[15] = '0' + (x.m/10)%10;
+    zBuf[16] = '0' + (x.m)%10;
+    zBuf[17] = ':';
+    s = (int)x.s;
+    zBuf[18] = '0' + (s/10)%10;
+    zBuf[19] = '0' + (s)%10;
+    zBuf[20] = 0;
+    if( x.Y<0 ){
+      zBuf[0] = '-';
+      sqlite3_result_text(context, zBuf, 20, SQLITE_TRANSIENT);
+    }else{
+      sqlite3_result_text(context, &zBuf[1], 19, SQLITE_TRANSIENT);
+    }
+  }
+}
